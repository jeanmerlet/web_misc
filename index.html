Problem 2

var sum = 0;
var fibNum_1 = 1;
var fibNum = 2;
while (fibNum < 4000000){
if (!(fibNum % 2)){
sum += fibNum;
}
fibNum += fibNum_1;
fibNum_1 = fibNum - fibNum_1;
}
console.log(sum);
